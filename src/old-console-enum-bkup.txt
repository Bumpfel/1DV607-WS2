public enum MenuOptions { 
		EditName("Edit name"), 
		EditPNr("Edit social security number"),
		CompactList("View compact list"),
		VerboseList("View verbose list"),
		;
		
		private String option;
	
		private MenuOptions(String str) {
			option = str;
		}
		
		public String getOption() {
			return option;
		}
	}
